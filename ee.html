def plot_data(df, cols=None, type=None):
    if type == 1:
        fig = plt.figure()
#         plt.figure(figsize=(18, 9))
        df[cols].boxplot()
        plt.title("Numerical variables in this dataset", fontsize=20)
        plt.savefig('Numerical.png')
    elif type == 2:
        fig = plt.figure()
        # ax = fig.add_subplot(1,3,i)
        var = cols[0]
        data = pd.concat([df['expire_flag'], df[var]], axis=1)
        # print(df[var].value_count:())
        # f, ax = plt.subplots(figsize=(8, 6))
        sns.countplot(x=var, hue='expire_flag', data=data)
        # fig.axis(ymin=0, ymax=800000)
        plt.savefig('Categorical.png')
    elif type == 3:
        fig = plt.figure()
        data_corr = df[cols].corr()
        # data.fillna(value=np.nan, inplace=True)
        sns.heatmap(data_corr, xticklabels=1, yticklabels=1)
        plt.title('Correlation Matric')
        plt.savefig('Correlation_Matric.png')
    elif type == 4:
        fig = plt.figure()
        sns.pointplot(x='model', y='auc', data=df)
        plt.title('Model Comparison')
        plt.savefig('Model_Comparison.png')



def load_data(filename):
    df = pd.read_csv(filename, header = 0, sep=',', dtype=col_types)
    # print(df[['hadm_id','expire_flag','religion']].groupby(['religion','expire_flag']).count())

    # print('\n', df.dtypes)

    ######################################### missing data
    features = pd.DataFrame({'a': df.apply(lambda x:sum(x.isnull())/df.shape[0])}).\
        sort_values('a')
    # print('missing value:')
    # print(features)
    features = features[features.a < 0.5]
    # print(features)
    df = df.loc[:, features.index]
    df.at[df[df['age'] > 200].index, 'age'] = 80
    # print(df[df['age']== 300].index)

    X = df.drop(['expire_flag','discharge_location','dob','admittime'], axis=1)
    y = df['expire_flag'].astype('category')

    ######################################### EDA
    full_cols = [
        '51279_itemid_min',
        '51277_itemid_min',
        '51265_itemid_min',
        '51250_itemid_min',
        'age',
        '51249_itemid_min',
        '51248_itemid_min',
        '51222_itemid_min',
        '51222_itemid_max',
        '51221_itemid_min',
        '51006_itemid_min',
        '50912_itemid_min'
    ]
    # print(list(col_types.keys()))
    # print(list(col_types.keys()).append('expire_flag'))
    plot_data(X, cols=['age', 'icu_no', '211_itemid_max'], type=1)
    plot_data(df, cols=['gender'], type=2)
    plot_data(df, cols=full_cols, type=3)

    ######################################### one-hot encoding, fill na, standard scale, feature selection
    X = pd.get_dummies(X).fillna(method='ffill').fillna(df.mean()).drop(['subject_id', 'hadm_id'], axis=1)
    names = X.columns
    X = pd.DataFrame(scale(X))
    X = StandardScaler().fit_transform(X)
    X = pd.DataFrame(X, columns=names)

    selectKBest = SelectKBest(mutual_info_classif, k=30)  # f_regression k
    X = selectKBest.fit_transform(X, y)
    X = pd.DataFrame(X, columns=names[selectKBest.get_support()])
    # print(':\n', [i for i in enumerate(names[selectKBest.get_support()])])
    # print(X)

    ######################################### imbalance data
    [print('Class {} has {} instances before oversampling'.format(label, count))
     for label, count in zip(*np.unique(y, return_counts=True))]

    kmeans_smote = KMeansSMOTE(
        kmeans_args={
            'n_clusters': 3
        },
        smote_args={
            'k_neighbors': 10
        }
    )
    X_resampled, y_resampled = kmeans_smote.fit_sample(X, y)
    X =pd.DataFrame(X, columns=names[selectKBest.get_support()])

    [print('Class {} has {} instances after oversampling'.format(label, count))
     for label, count in zip(*np.unique(y_resampled, return_counts=True))]

    return X_resampled, y_resampled, X


def plot_feature_importance(model, feature_names):
    importance = model.feature_importances_
    std = np.std([tree.feature_importances_ for tree in model.estimators_], axis=0)
    indices = np.argsort(importance)[::-1]
    # Print the feature ranking
    print("Feature ranking:")

    for f in range(X.shape[1]):
        print("%d. %s (%f)" % (f + 1, feature_names[indices[f]], importance[indices[f]]))

    # Plot the feature importance of the forest
    plt.figure()
    plt.title("Feature importance")
    plt.bar(range(X.shape[1]), importance[indices],
            color="r", yerr=std[indices], align="center")
    plt.xticks(range(X.shape[1]), indices)
    plt.xlim([-1, X.shape[1]])
    plt.show()
